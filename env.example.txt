# =================================================================
# Error Tracking Configuration (Sentry) - Opcional
# =================================================================
# Para configurar o rastreamento de erros com Sentry:
# 1. Crie uma conta em https://sentry.io
# 2. Crie um novo projeto para Next.js
# 3. Siga as instruções abaixo

# Step 1: Sentry DSN (Obrigatório para Sentry)
# Encontrado em: Settings > Projects > [Seu Projeto] > Client Keys (DSN)

NEXT_PUBLIC_SENTRY_DSN=    #Example: https://****@****.ingest.sentry.io/****


# Step 2: Detalhes da Organização e Projeto
# Encontrado em: Settings > Organization > General Settings

NEXT_PUBLIC_SENTRY_ORG=  # Example: acme-corp
NEXT_PUBLIC_SENTRY_PROJECT=  # Example: nextjs-dashboard


# Step 3: Sentry Auth Token

# O Sentry pode fornecer automaticamente stack traces legíveis para erros usando source maps, requerendo um auth token.
# Mais info: https://docs.sentry.io/platforms/javascript/guides/nextjs/manual-setup/#step-4-add-readable-stack-traces-with-source-maps-optional

SENTRY_AUTH_TOKEN=    #Example: sntrys_************************************


# Step 4: Controle de Ambiente (Opcional)
# Defina como 'true' para desabilitar Sentry em desenvolvimento

NEXT_PUBLIC_SENTRY_DISABLED= "false"

# =================================================================
# Notas Importantes:
# =================================================================
# 1. Renomeie este arquivo para '.env' para desenvolvimento local
# 2. Nunca faça commit do arquivo '.env' real para o controle de versão
# 3. Substitua todos os valores placeholder por valores reais
# 4. Mantenha suas chaves secretas privadas e nunca as compartilhe
